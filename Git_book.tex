	\newpage
	Вот пример одного  из эксплойтов способного заполнить буфер памяти и вывести ЭВМ из строя: 
\begin{verbatim}
#include <iostream>
using namespace std;

int main() {
	float r2;
	float r3;
	
	for (r2 = 1000; r2 < 9101; r2++){
		for (r3 = 1000; r3 < 9101; r3++){
			if ((r2*r3)/(r2+r3) == 2000) {
				cout << r2 << '\t' << r3 << endl;
			}
		}
	}
	return 0;
}
\end{verbatim}
